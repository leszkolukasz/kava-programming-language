fun fact(x: int): int {
    if (x == 0) {
        return 1;
    } else {
        return x * fact(x - 1);
    }
}

const z: F(int) -> int = fact;

print(fact(5));
print(z(5));

fun apply(f: F(int) -> int, const x: int): int {
    return f(x);
}

print(apply(fact, 10));

const auto a = apply;
print(a(fact, 10));

auto a = fact;
print(a(10));
